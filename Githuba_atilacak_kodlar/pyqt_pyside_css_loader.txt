from PyQt5.QtCore import*
from PyQt5.QtWidgets import*
from PyQt5.QtGui import*
import sys as _s

def css_loader():
    file = QFile('fun_app_css.qss')
        
    if file.open(QIODevice.ReadOnly | QIODevice.Text):
        css = QTextStream(file)
        css.setCodec('utf-8')
        readed_css = css.readAll()
        file.close()

    return readed_css

class İnit_Ui(QMainWindow):
    def __init__(self):
        super().__init__()

        self.setObjectName('window')
        self.setStyleSheet(css_loader())
        
if __name__=="__main__":
    sp = QApplication(_s.argv)
    sw = İnit_Ui()
    sw.show()
    _s.exit(sp.exec_())
